## Description

Given an integer array `nums`, find the contiguous subarray within an array (containing at least one number) which has the largest product.

**Example 1:**

```
Input: [2,3,-2,4]
Output: 6
Explanation: [2,3] has the largest product 6.
```

**Example 2:**

```
Input: [-2,0,-1]
Output: 0
Explanation: The result cannot be 2, because [-2,-1] is not a subarray.
```

```java
class Solution {
    public int maxProduct(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        // max value of per position        
        int[] maxArr = new int[n];
        maxArr[0] = nums[0];
        
        // min value of position: process negative number
        int[] minArr = new int[n];
        minArr[0] = nums[0];
        
        // answer
        int answer = nums[0];        
        for (int i = 1; i < n; i++) {
            int a = nums[i];
            int b = nums[i] * maxArr[i - 1];
            int c = nums[i] * minArr[i - 1];
            
            // max value of current position
            int max = Math.max(a, Math.max(b, c));    
            maxArr[i] = max;            
            
            // min value of current position 
            minArr[i] = Math.min(a, Math.min(b, c));
            
            // compare and get max
            answer = Math.max(answer, max);
        }
        
        return answer;
    }
}
```

class Solution {
    public int maxProduct(int[] nums) {
        int n = nums.length;
        if (n == 0) {
            return 0;
        }
        
        int prevMax = nums[0];                
        int prevMin = nums[0];
        
        // answer
        int answer = nums[0];        
        for (int i = 1; i < n; i++) {
            int a = nums[i];
            int b = nums[i] * prevMax;
            int c = nums[i] * prevMin;
            
            // max value of current position
            int max = Math.max(a, Math.max(b, c));    
            prevMax = max;            
            
            // min value of current position 
            prevMin = Math.min(a, Math.min(b, c));
            
            // compare and get max
            answer = Math.max(answer, max);
        }
        
        return answer;
    }
}
